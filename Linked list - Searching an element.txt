Write a Program to search an element from the list.

Input Format

Read elements upto -1

Constraints

1<=n<=10^7

Output Format

print output

1
2
3
4
-1
3

3 is present in the list

1

-1

Sample Input 2

23
11

53

88

88 not present in the list

Code:

class Node:
    def __init__(self,data):
        self.data=data
        self.next=None

class LinkedList:
    def __init__(self):
        self.head=self.tail=None
    
    def create(self,x):
        n=Node(x)
        if(self.head==None):
            self.head=self.tail=n
        else:
            self.tail.next=n
            self.tail=n
      
    
    def search(self,k):
        c=0
        temp=self.head
        while temp!=None:
            temp=temp.next
            c=c+1
        temp=self.head
        for i in range(0,c):
            if k==temp.data:
                print(k,"is present in the list")
                return
            else:
                temp=temp.next
        print(k,"not present in the list")
        return    
            
l1=LinkedList()
while True:
    x=int(input())
    if x==-1:
        break
    else:
        l1.create(x)
k=int(input())
l1.search(k)
